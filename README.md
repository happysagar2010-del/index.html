        // Download as image
        async function downloadAsImage() {
            const loadingOverlay = document.getElementById('loadingOverlay');
            const paperPreview = document.querySelector('.paper-preview');
            
            if (!paperPreview) return;
            
            loadingOverlay.style.display = 'flex';
            
            try {
                await new Promise(resolve => setTimeout(resolve, 500));
                
                const canvas = await html2canvas(paperPreview, {
                    scale: 3,
                    backgroundColor: paperPreview.style.background || '#fefefe',
                    useCORS: true,
                    logging: false
                });
                
                const link = document.createElement('a');
                const timestamp = new Date().toISOString().slice(0,19).replace(/[:T]/g, '-');
                link.download = `handwriting-${timestamp}.png`;
                link.href = canvas.toDataURL('image/png', 1.0);
                
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                
                const successMessage = document.getElementById('successMessage');
                successMessage.style.display = 'flex';
                
                setTimeout(() => {
                    successMessage.style.display = 'none';
                }, 3000);
                
            } catch (error) {
                console.error('Error generating image:', error);
                alert('There was an error generating the image. Please try again.');
            } finally {
                loadingOverlay.style.display = 'none';
            }
        }
